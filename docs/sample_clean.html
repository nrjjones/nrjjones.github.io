Separate names from single field

df <- df %>% 
  mutate(
    first = str_extract(fullname, ", [A-Za-z]*"),
    first = str_remove(first, ", "),
    first = str_extract(fullname, "[A-Za-z], "),
    last = str_remove(last, ", ")
